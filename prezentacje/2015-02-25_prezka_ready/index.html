Liczby (int, float)

a = 10
b = 5
a * b  # 50
b / a  # 0.5
a + b  # 15
b - a  # -5
a ** b # 100000

Bool (bool)

a = True
b = False
a and b  # False
a or b   # True
a or (b and a)  # ???

Napisy (str)

a = 'Ala ma kota'
b = 'kot ma Alę'
a + ', a ' + b  # Ala ma kota, a kot ma Alę
a[1]     # l
a[2:5]   # a m
a[2:9:3] # aao
a[::-1]  # atok am alA

Krotki (tuple)

a = (1, 'a', 40, 'miauuu...')
b = (2, 'b', 80, 'miauuu, miauuu...')
a + b  # (1, 'a', 40, 'miauuu...', 2, 'b', 80, 'miauuu, miauuu...')
a[1]    # a
b[1:3]  # ('b', 80)

# ... Jak w napisach ...

Pętla for x in y

my_numbers = [1, 2, 3, 4]

for number in my_numbers:
    print('I like number: ', number)

name = 'Grześ'

for letter in name:
    print(letter)

my_name_letters = ('G', 'r', 'z', 'e', 'ś')

for letter in my_name_letters:
    print(letter)
